import { useEffect, useState } from "react";
import { useParams } from "react-router-dom";
import { apiRequest } from "../api/client";
import Navbar from "../components/Navbar";
import Footer from "../components/Footer";
type Price={karat:string;makingCharge:number;totalCost:number;gst:number;finalPrice:number;grossWeight:number;netWeight:number};type Review={_id:string;rating:number;text:string;user?:{name:string}}; type Product={SKU:string;Title:string;Description:string;Category:string;colors?:string[];size_options?:string[];prices:Price[];[key:string]:any};
export default function ProductDetails(){const {slug=""}=useParams();const [product,setProduct]=useState<Product|null>(null);const [karat,setKarat]=useState("14kt");const [color,setColor]=useState("");const [size,setSize]=useState("");const [message,setMessage]=useState("");const [reviews,setReviews]=useState<Review[]>([]);const [coupons,setCoupons]=useState<{code:string;scope:string}[]>([]);const [rating,setRating]=useState(5);const [reviewText,setReviewText]=useState("");useEffect(()=>{apiRequest<Product>(`/products/${slug}`).then(p=>{setProduct(p);return apiRequest<Review[]>(`/reviews/${p.SKU}`)}).then(setReviews);apiRequest<{code:string;scope:string}[]>(`/coupons/eligible/${slug}`).then(setCoupons).catch(()=>setProduct(null))},[slug]);if(!product)return <><Navbar onSearchChange={()=>{}} activeCategory="All" onCategoryChange={()=>{}}/><main className="p-10">Loading product…</main></>;const price=product.prices.find(p=>p.karat===karat)||product.prices[0];const submitReview=async()=>{try{const r=await apiRequest<Review>(`/reviews/${product.SKU}`,{method:"POST",body:JSON.stringify({rating,text:reviewText})});setMessage("Review submitted for moderation");setReviewText("")}catch(e:any){setMessage(e.message)}};const add=async()=>{try{await apiRequest("/cart",{method:"POST",body:JSON.stringify({productId:product.SKU,karat,color,size,quantity:1})});setMessage("Added to cart")}catch(e:any){setMessage(e.message)}};const wish=async()=>{try{await apiRequest("/wishlist",{method:"POST",body:JSON.stringify({productId:product.SKU,karat})});setMessage("Added to wishlist")}catch(e:any){setMessage(e.message)}};return <><Navbar onSearchChange={()=>{}} activeCategory="All" onCategoryChange={()=>{}}/><main className="mx-auto grid max-w-6xl gap-8 p-6 md:grid-cols-2"><img src={product["image_link-1"]} alt={product.Title} className="w-full"/><section><p>{product.Category}</p><h1 className="text-3xl">{product.Title}</h1><p>{product.Description}</p><div className="my-4 flex gap-2">{["9kt","14kt","18kt"].map(k=><button key={k} onClick={()=>setKarat(k)} className="border p-2">{k.toUpperCase()}</button>)}</div>{product.colors?.length?<select value={color} onChange={e=>setColor(e.target.value)}><option value="">Select colour</option>{product.colors.map(c=><option key={c}>{c}</option>)}</select>:null}{product.size_options?.length?<select value={size} onChange={e=>setSize(e.target.value)}><option value="">Select size</option>{product.size_options.map(s=><option key={s}>{s}</option>)}</select>:null}<p className="mt-4 text-2xl">₹{Math.round(price.finalPrice).toLocaleString("en-IN")}</p><p>Making charge: ₹{Math.round(price.makingCharge).toLocaleString("en-IN")} · GST: ₹{Math.round(price.gst).toLocaleString("en-IN")}</p><p>Gross weight: {price.grossWeight}g · Net weight: {price.netWeight}g</p><div className="mt-6 flex gap-3"><button onClick={add} className="bg-black px-5 py-3 text-white">Add to Cart</button><button onClick={wish} className="border px-5 py-3">Add to Wishlist</button></div><p>{message}</p><div className="mt-4">{coupons.map(c=><span className="mr-2 border p-1" key={c.code}>{c.code}</span>)}</div><section className="mt-10 border-t pt-6"><h2 className="text-xl">Reviews</h2>{reviews.map(r=><article key={r._id} className="my-3"><b>{r.user?.name||"Customer"} · {r.rating}/5</b><p>{r.text}</p></article>)}<div className="mt-4"><select value={rating} onChange={e=>setRating(Number(e.target.value))}>{[1,2,3,4,5].map(n=><option key={n}>{n}</option>)}</select><textarea value={reviewText} onChange={e=>setReviewText(e.target.value)} placeholder="Write a review" className="block w-full border p-2"/><button onClick={submitReview} className="mt-2 border p-2">Submit review</button></div></section></section></main><Footer onCategoryChange={()=>{}}/></>}
